性感的订阅语法

let todo: RxAVObject = new RxAVObject('RxTodo');
todo.set('title', '开会');
todo.set('time', '2016-12-03');
todo.set('reminder', new Date());
todo.save().subscribe(success => {
    if (success) console.log(todo.objectId);
});
